--ADMIN
CREATE USER C##ADMIN IDENTIFIED BY 1234;
GRANT DBA TO C##ADMIN;

--TEACHER
CREATE USER C##TEACHER IDENTIFIED BY 1234;
GRANT SELECT ON C##ADMIN.STUDENT TO C##TEACHER;
GRANT CONNECT TO C##TEACHER;

--TEACHERS VIEW THEIR STUDENT
CREATE OR REPLACE VIEW teacher_1_students AS
SELECT 
    t.teacher_id, 
    t.fname AS teacher_fname, 
    t.lname AS teacher_lname, 
    s.student_id, 
    s.gender,
    s.DOB,
    s.email,
    s.phone_number,
    s.address,
    s.GPA,
    s.status,
    s.enrollment_year,
    s.fname AS student_fname, 
    s.lname AS student_lname, 
    c.course_id, 
    c.class_id
FROM 
    Teacher t
JOIN 
    Class c ON t.teacher_id = c.teacher_id
JOIN 
    Attends a ON c.course_id = a.course_id AND c.class_id = a.class_id
JOIN 
    Student s ON a.student_id = s.student_id
WHERE 
    t.teacher_id = 1;
GRANT SELECT ON C##ADMIN.teacher_1_students TO C##TEACHER;

DROP VIEW C##ADMIN.teacher_1_students;


--TEACHERS TIMETABLE
CREATE OR REPLACE VIEW teacher_1_timetable AS
SELECT 
    t.teacher_id,
    t.fname || ' ' || t.lname AS teacher_name,
    c.class_id,
    co.course_id,
    co.course_name,
    co.course_description,
    ti.room,
    ti.building,
    ti.day_of_week,
    ti.start_time,
    ti.end_time
FROM 
    Teacher t
JOIN 
    Class c ON t.teacher_id = c.teacher_id
JOIN 
    Course co ON c.course_id = co.course_id
JOIN
    Time ti ON c.course_id = ti.course_id AND c.class_id = ti.class_id
WHERE 
    t.teacher_id = 1;
GRANT SELECT ON C##ADMIN.TEACHER_1_TIMETABLE TO C##TEACHER;

DROP VIEW C##ADMIN.TEACHER_1_TIMETABLE;

--TEACHERS UPDATE THEIR PERSONAL INFORMATION
CREATE OR REPLACE VIEW teacher_1_update_info AS
SELECT * FROM C##ADMIN.teacher where teacher_id = 1;
GRANT UPDATE ON teacher_1_update_info TO C##TEACHER;

--TEACHERS VIEW AND UPATE THEIR CLASS SCORE AND GRADE
CREATE OR REPLACE VIEW teacher_1_view_update_exam AS
SELECT 
    c.class_id,   
    co.course_id,   
    e.exam_id,
    s.student_id,
    s.fname || ' ' || s.lname AS student_name,
    e.exam_date,
    e.exam_type,
    e.grading_date,
    co.course_name,
    tk.score,
    tk.grade
    
FROM 
    Teacher t
JOIN 
    Class c ON t.teacher_id = c.teacher_id
JOIN 
    Course co ON c.course_id = co.course_id
JOIN 
    Exam e ON co.course_id = e.exam_id AND t.teacher_id = e.teacher_id
JOIN
    Takes tk ON e.exam_id = tk.exam_id
JOIN
    Student s ON tk.student_id = s.student_id
WHERE 
    t.teacher_id = 1;

GRANT SELECT ON C##ADMIN.teacher_1_view_update_exam TO C##TEACHER;
GRANT UPDATE ON C##ADMIN.teacher_1_view_update_exam TO C##TEACHER;

DROP VIEW teacher_1_view_update_exam;
